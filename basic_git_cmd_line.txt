git add .          //Stages all files, 
git add test.cpp   //stages test.cpp
git status         //lists all modifications or added files
git commit -m "commit message"   //saves the changes to the local git repo
git push          //pushes saved changes to server repo
git pull          //pulls saved changes from server repo, this is when conflicts occur

git branch        //lists all branches
git branch new_branch  //creates a new branch named new_branch with a copy from all the code that is currently checked out
